Task_3:
SELECT *
FROM employees WHERE (department_id = 50 OR department_id = 80)

Task_5:

Task_6:
SELECT department_id, salary, COUNT(*) as employee_count
FROM employees GROUP BY department_id, salary
HAVING COUNT(*) > 1;
